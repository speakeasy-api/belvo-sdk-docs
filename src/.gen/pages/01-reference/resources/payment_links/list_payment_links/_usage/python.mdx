import belvo
import dateutil.parser
from belvo.models import components, operations

s = belvo.Belvo(
    security=components.Security(
        password="",
        username="",
    ),
)

req = operations.ListPaymentLinksRequest(
    created_at=dateutil.parser.parse('2022-05-05').date(),
    created_at_gt=dateutil.parser.parse('2022-05-05').date(),
    created_at_gte=dateutil.parser.parse('2022-05-04').date(),
    created_at_lt=dateutil.parser.parse('2022-04-01').date(),
    created_at_lte=dateutil.parser.parse('2022-03-30').date(),
    created_at_range=[
        dateutil.parser.parse('2').date(),
        dateutil.parser.parse('0').date(),
        dateutil.parser.parse('2').date(),
        dateutil.parser.parse('2').date(),
        dateutil.parser.parse('-').date(),
        dateutil.parser.parse('0').date(),
        dateutil.parser.parse('3').date(),
        dateutil.parser.parse('-').date(),
        dateutil.parser.parse('0').date(),
        dateutil.parser.parse('3').date(),
        dateutil.parser.parse(',').date(),
        dateutil.parser.parse('2').date(),
        dateutil.parser.parse('0').date(),
        dateutil.parser.parse('2').date(),
        dateutil.parser.parse('2').date(),
        dateutil.parser.parse('-').date(),
        dateutil.parser.parse('0').date(),
        dateutil.parser.parse('5').date(),
        dateutil.parser.parse('-').date(),
        dateutil.parser.parse('0').date(),
        dateutil.parser.parse('4').date(),
    ],
    ordering=components.OrderingPaymentLinks.MINUS_CREATED_AT,
    page=1,
    page_size=100,
    search='trai',
    status=components.StatusPaymentLinks.ACTIVE,
)

res = s.payment_links.list_payment_links(req)
