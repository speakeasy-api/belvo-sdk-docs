  const res = await sdk.paymentInstitutions.listPaymentInstitutions({

      new RFCDate("0"),
      new RFCDate("3"),

      new RFCDate("2"),

      new RFCDate("5"),
      new RFCDate("-"),
      new RFCDate("0"),
      new RFCDate("4"),
    ],
    page: 1,
    pageSize: 100,
  });

  if (res.statusCode == 200) {
    // handle response

})();
